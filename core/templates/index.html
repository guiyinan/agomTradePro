快速开始

# 1. 激活虚拟环境
agomsaaf\Scripts\activate

# 2. 创建管理员
python manage.py createsuperuser

# 3. 启动服务器
python manage.py runserver

# 4. 访问系统
# 首页: http://127.0.0.1:8000/
# 管理后台: http://127.0.0.1:8000/admin/
# API文档: http://127.0.0.1:8000/api/docs/
